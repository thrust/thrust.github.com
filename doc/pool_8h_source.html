170  // this was originally a forward list, but I made it a doubly linked list
171  // because that way deallocation when not caching is faster and doesn't require
172  // traversal of a linked list (it's still a forward list for the cached list,
173  // because allocation from that list already traverses)
174  //
175  // TODO: investigate whether it's better to have this be a doubly-linked list
176  // with fast do_deallocate when !m_options.cache_oversized, or to have this be
177  // a forward list and require traversal in do_deallocate
178  //
179  // I assume that it is better this way, but the additional pointer could
180  // potentially hurt? these are supposed to be oversized and/or overaligned,
181  // so they are kinda memory intensive already
182  struct oversized_block_descriptor
183  {
184  std::size_t size;
185  std::size_t alignment;
186  oversized_block_descriptor_ptr prev;
187  oversized_block_descriptor_ptr next;
188  oversized_block_descriptor_ptr next_cached;
189  };

215  void release()
216  {
217  // reset the buckets
218  for (std::size_t i = 0; i < m_pools.size(); ++i)
219  {
220  thrust::raw_reference_cast(m_pools[i]).free_list = block_descriptor_ptr();
221  thrust::raw_reference_cast(m_pools[i]).previous_allocated_count = 0;
222  }
223 
224  // deallocate memory allocated for the buckets
225  while (detail::pointer_traits<chunk_descriptor_ptr>::get(m_allocated))
226  {
227  chunk_descriptor_ptr alloc = m_allocated;
228  m_allocated = thrust::raw_reference_cast(*m_allocated).next;
229 
230  void_ptr p = static_cast<void_ptr>(
231  static_cast<char_ptr>(
232  static_cast<void_ptr>(alloc)
233  ) - thrust::raw_reference_cast(*alloc).size
234  );
235  m_upstream->do_deallocate(p, thrust::raw_reference_cast(*alloc).size + sizeof(chunk_descriptor), m_options.alignment);
236  }
237 
238  // deallocate cached oversized/overaligned memory
239  while (detail::pointer_traits<oversized_block_descriptor_ptr>::get(m_oversized))
240  {
241  oversized_block_descriptor_ptr alloc = m_oversized;
242  m_oversized = thrust::raw_reference_cast(*m_oversized).next;
243 
244  void_ptr p = static_cast<void_ptr>(
245  static_cast<char_ptr>(
246  static_cast<void_ptr>(alloc)
247  ) - thrust::raw_reference_cast(*alloc).size
248  );
249  m_upstream->do_deallocate(p, thrust::raw_reference_cast(*alloc).size + sizeof(oversized_block_descriptor), thrust::raw_reference_cast(*alloc).alignment);
250  }
251 
252  m_cached_oversized = oversized_block_descriptor_ptr();
253  }

255  THRUST_NODISCARD virtual void_ptr do_allocate(std::size_t bytes, std::size_t alignment = THRUST_MR_DEFAULT_ALIGNMENT) override
256  {
257  bytes = (std::max)(bytes, m_options.smallest_block_size);
258  assert(detail::is_power_of_2(alignment));
259 
260  // an oversized and/or overaligned allocation requested; needs to be allocated separately
261  if (bytes > m_options.largest_block_size || alignment > m_options.alignment)
262  {
263  if (m_options.cache_oversized)
264  {
265  oversized_block_descriptor_ptr ptr = m_cached_oversized;
266  oversized_block_descriptor_ptr * previous = &m_cached_oversized;
267  while (detail::pointer_traits<oversized_block_descriptor_ptr>::get(ptr))
268  {
269  oversized_block_descriptor desc = *ptr;
270  bool is_good = desc.size >= bytes && desc.alignment >= alignment;
271 
272  // if the size is bigger than the requested size by a factor
273  // bigger than or equal to the specified cutoff for size,
274  // allocate a new block
275  if (is_good)
276  {
277  std::size_t size_factor = desc.size / bytes;
278  if (size_factor >= m_options.cached_size_cutoff_factor)
279  {
280  is_good = false;
281  }
282  }
283 
284  // if the alignment is bigger than the requested one by a factor
285  // bigger than or equal to the specified cutoff for alignment,
286  // allocate a new block
287  if (is_good)
288  {
289  std::size_t alignment_factor = desc.alignment / alignment;
290  if (alignment_factor >= m_options.cached_alignment_cutoff_factor)
291  {
292  is_good = false;
293  }
294  }
295 
296  if (is_good)
297  {
298  if (previous != &m_cached_oversized)
299  {
300  oversized_block_descriptor previous_desc = **previous;
301  previous_desc.next_cached = desc.next_cached;
302  **previous = previous_desc;
303  }
304  else
305  {
306  m_cached_oversized = desc.next_cached;
307  }
308 
309  desc.next_cached = oversized_block_descriptor_ptr();
310  *ptr = desc;
311 
312  return static_cast<void_ptr>(
313  static_cast<char_ptr>(
314  static_cast<void_ptr>(ptr)
315  ) - desc.size
316  );
317  }
318 
319  previous = &thrust::raw_reference_cast(*ptr).next_cached;
320  ptr = *previous;
321  }
322  }
323 
324  // no fitting cached block found; allocate a new one that's just up to the specs
325  void_ptr allocated = m_upstream->do_allocate(bytes + sizeof(oversized_block_descriptor), alignment);
326  oversized_block_descriptor_ptr block = static_cast<oversized_block_descriptor_ptr>(
327  static_cast<void_ptr>(
328  static_cast<char_ptr>(allocated) + bytes
329  )
330  );
331 
332  oversized_block_descriptor desc;
333  desc.size = bytes;
334  desc.alignment = alignment;
335  desc.prev = oversized_block_descriptor_ptr();
336  desc.next = m_oversized;
337  desc.next_cached = oversized_block_descriptor_ptr();
338  *block = desc;
339  m_oversized = block;
340 
341  if (detail::pointer_traits<oversized_block_descriptor_ptr>::get(desc.next))
342  {
343  oversized_block_descriptor next = *desc.next;
344  next.prev = block;
345  *desc.next = next;
346  }
347 
348  return allocated;
349  }
350 
351  // the request is NOT for oversized and/or overaligned memory
352  // allocate a block from an appropriate bucket
353  std::size_t bytes_log2 = thrust::detail::log2_ri(bytes);
354  std::size_t bucket_idx = bytes_log2 - m_smallest_block_log2;
355  pool & bucket = thrust::raw_reference_cast(m_pools[bucket_idx]);
356 
357  bytes = static_cast<std::size_t>(1) << bytes_log2;
358 
359  // if the free list of the bucket has no elements, allocate a new chunk
360  // and split it into blocks pushed to the free list
361  if (!detail::pointer_traits<block_descriptor_ptr>::get(bucket.free_list))
362  {
363  std::size_t n = bucket.previous_allocated_count;
364  if (n == 0)
365  {
366  n = m_options.min_blocks_per_chunk;
367  if (n < (m_options.min_bytes_per_chunk >> bytes_log2))
368  {
369  n = m_options.min_bytes_per_chunk >> bytes_log2;
370  }
371  }
372  else
373  {
374  n = n * 3 / 2;
375  if (n > (m_options.max_bytes_per_chunk >> bytes_log2))
376  {
377  n = m_options.max_bytes_per_chunk >> bytes_log2;
378  }
379  if (n > m_options.max_blocks_per_chunk)
380  {
381  n = m_options.max_blocks_per_chunk;
382  }
383  }
384 
385  std::size_t descriptor_size = (std::max)(sizeof(block_descriptor), m_options.alignment);
386  std::size_t block_size = bytes + descriptor_size;
387  block_size += m_options.alignment - block_size % m_options.alignment;
388  std::size_t chunk_size = block_size * n;
389 
390  void_ptr allocated = m_upstream->do_allocate(chunk_size + sizeof(chunk_descriptor), m_options.alignment);
391  chunk_descriptor_ptr chunk = static_cast<chunk_descriptor_ptr>(
392  static_cast<void_ptr>(
393  static_cast<char_ptr>(allocated) + chunk_size
394  )
395  );
396 
397  chunk_descriptor chunk_desc;
398  chunk_desc.size = chunk_size;
399  chunk_desc.next = m_allocated;
400  *chunk = chunk_desc;
401  m_allocated = chunk;
402 
403  for (std::size_t i = 0; i < n; ++i)
404  {
405  block_descriptor_ptr block = static_cast<block_descriptor_ptr>(
406  static_cast<void_ptr>(
407  static_cast<char_ptr>(allocated) + block_size * i + bytes
408  )
409  );
410 
411  block_descriptor block_desc;
412  block_desc.next = bucket.free_list;
413  *block = block_desc;
414  bucket.free_list = block;
415  }
416  }
417 
418  // allocate a block from the front of the bucket's free list
419  block_descriptor_ptr block = bucket.free_list;
420  bucket.free_list = thrust::raw_reference_cast(*block).next;
421  return static_cast<void_ptr>(
422  static_cast<char_ptr>(
423  static_cast<void_ptr>(block)
424  ) - bytes
425  );
426  }

428  virtual void do_deallocate(void_ptr p, std::size_t n, std::size_t alignment = THRUST_MR_DEFAULT_ALIGNMENT) override
429  {
430  n = (std::max)(n, m_options.smallest_block_size);
431  assert(detail::is_power_of_2(alignment));
432 
433  // verify that the pointer is at least as aligned as claimed
434  assert(reinterpret_cast<detail::intmax_t>(detail::pointer_traits<void_ptr>::get(p)) % alignment == 0);
435 
436  // the deallocated block is oversized and/or overaligned
437  if (n > m_options.largest_block_size || alignment > m_options.alignment)
438  {
439  oversized_block_descriptor_ptr block = static_cast<oversized_block_descriptor_ptr>(
440  static_cast<void_ptr>(
441  static_cast<char_ptr>(p) + n
442  )
443  );
444 
445  oversized_block_descriptor desc = *block;
446 
447  if (m_options.cache_oversized)
448  {
449  desc.next_cached = m_cached_oversized;
450  *block = desc;
451  m_cached_oversized = block;
452 
453  return;
454  }
455 
456  if (!detail::pointer_traits<oversized_block_descriptor_ptr>::get(desc.prev))
457  {
458  assert(m_oversized == block);
459  m_oversized = desc.next;
460  }
461  else
462  {
463  oversized_block_descriptor prev = *desc.prev;
464  assert(prev.next == block);
465  prev.next = desc.next;
466  *desc.prev = prev;
467  }
468 
469  if (detail::pointer_traits<oversized_block_descriptor_ptr>::get(desc.next))
470  {
471  oversized_block_descriptor next = *desc.next;
472  assert(next.prev == block);
473  next.prev = desc.prev;
474  *desc.next = next;
475  }
476 
477  m_upstream->do_deallocate(p, desc.size + sizeof(oversized_block_descriptor), desc.alignment);
478 
479  return;
480  }
481 
482  // push the block to the front of the appropriate bucket's free list
483  std::size_t n_log2 = thrust::detail::log2_ri(n);
484  std::size_t bucket_idx = n_log2 - m_smallest_block_log2;
485  pool & bucket = thrust::raw_reference_cast(m_pools[bucket_idx]);
486 
487  n = static_cast<std::size_t>(1) << n_log2;
488 
489  block_descriptor_ptr block = static_cast<block_descriptor_ptr>(
490  static_cast<void_ptr>(
491  static_cast<char_ptr>(p) + n
492  )
493  );
494 
495  block_descriptor desc;
496  desc.next = bucket.free_list;
497  *block = desc;
498  bucket.free_list = block;
499  }
