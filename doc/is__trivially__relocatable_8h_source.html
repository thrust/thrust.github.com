242 // There is no way to actually detect the libstdc++ version; __GLIBCXX__
243 // is always set to the date of libstdc++ being packaged, not the release
244 // day or version. This means that we can't detect the libstdc++ version,
245 // except when compiling with GCC.
246 //
247 // Therefore, for the best approximation of is_trivially_copyable, we need to
248 // handle three distinct cases:
249 // 1) GCC above 5, or another C++11 compiler not using libstdc++: use the
250 // standard trait directly.
251 // 2) A C++11 compiler using libstdc++ that provides the intrinsic: use the
252 // intrinsic.
253 // 3) Any other case (essentially: compiling without C++11): has_trivial_assign.
254 
255 #ifndef __has_feature
256  #define __has_feature(x) 0
257 #endif

263  #if THRUST_CPP_DIALECT >= 2011
264  #if defined(__GLIBCXX__) && __has_feature(is_trivially_copyable)
265  __is_trivially_copyable(T)
266  #elif THRUST_HOST_COMPILER == THRUST_HOST_COMPILER_GCC && THRUST_GCC_VERSION >= 50000
267  std::is_trivially_copyable<T>::value
268  #else
269  has_trivial_assign<T>::value
270  #endif
