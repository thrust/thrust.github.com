314  THRUST_NODISCARD virtual void_ptr do_allocate(std::size_t bytes, std::size_t alignment = THRUST_MR_DEFAULT_ALIGNMENT) THRUST_OVERRIDE
315  {
316  bytes = (std::max)(bytes, m_options.smallest_block_size);
317  assert(detail::is_power_of_2(alignment));
318 
319  // an oversized and/or overaligned allocation requested; needs to be allocated separately
320  if (bytes > m_options.largest_block_size || alignment > m_options.alignment)
321  {
322  oversized_block_descriptor oversized;
323  oversized.size = bytes;
324  oversized.alignment = alignment;
325 
326  if (m_options.cache_oversized && !m_cached_oversized.empty())
327  {
328  typename oversized_block_vector::iterator it = thrust::lower_bound(
329  thrust::seq,
330  m_cached_oversized.begin(),
331  m_cached_oversized.end(),
332  oversized);
333 
334  // if the size is bigger than the requested size by a factor
335  // bigger than or equal to the specified cutoff for size,
336  // allocate a new block
337  if (it != m_cached_oversized.end())
338  {
339  std::size_t size_factor = (*it).size / bytes;
340  if (size_factor >= m_options.cached_size_cutoff_factor)
341  {
342  it = m_cached_oversized.end();
343  }
344  }
345 
346  if (it != m_cached_oversized.end() && (*it).alignment < alignment)
347  {
348  it = find_if(it + 1, m_cached_oversized.end(), matching_alignment(alignment));
349  }
350 
351  // if the alignment is bigger than the requested one by a factor
352  // bigger than or equal to the specified cutoff for alignment,
353  // allocate a new block
354  if (it != m_cached_oversized.end())
355  {
356  std::size_t alignment_factor = (*it).alignment / alignment;
357  if (alignment_factor >= m_options.cached_alignment_cutoff_factor)
358  {
359  it = m_cached_oversized.end();
360  }
361  }
362 
363  if (it != m_cached_oversized.end())
364  {
365  oversized.pointer = (*it).pointer;
366  m_cached_oversized.erase(it);
367  return oversized.pointer;
368  }
369  }
370 
371  // no fitting cached block found; allocate a new one that's just up to the specs
372  oversized.pointer = m_upstream->do_allocate(bytes, alignment);
373  m_oversized.push_back(oversized);
374 
375  return oversized.pointer;
376  }
377 
378  // the request is NOT for oversized and/or overaligned memory
379  // allocate a block from an appropriate bucket
380  std::size_t bytes_log2 = thrust::detail::log2_ri(bytes);
381  std::size_t bucket_idx = bytes_log2 - m_smallest_block_log2;
382  pool & bucket = m_pools[bucket_idx];
383 
384  // if the free list of the bucket has no elements, allocate a new chunk
385  // and split it into blocks pushed to the free list
386  if (bucket.free_blocks.empty())
387  {
388  std::size_t bucket_size = static_cast<std::size_t>(1) << bytes_log2;
389 
390  std::size_t n = bucket.previous_allocated_count;
391  if (n == 0)
392  {
393  n = m_options.min_blocks_per_chunk;
394  if (n < (m_options.min_bytes_per_chunk >> bytes_log2))
395  {
396  n = m_options.min_bytes_per_chunk >> bytes_log2;
397  }
398  }
399  else
400  {
401  n = n * 3 / 2;
402  if (n > (m_options.max_bytes_per_chunk >> bytes_log2))
403  {
404  n = m_options.max_bytes_per_chunk >> bytes_log2;
405  }
406  if (n > m_options.max_blocks_per_chunk)
407  {
408  n = m_options.max_blocks_per_chunk;
409  }
410  }
411 
412  bytes = n << bytes_log2;
413 
414  assert(n >= m_options.min_blocks_per_chunk);
415  assert(n <= m_options.max_blocks_per_chunk);
416  assert(bytes >= m_options.min_bytes_per_chunk);
417  assert(bytes <= m_options.max_bytes_per_chunk);
418 
419  chunk_descriptor allocated;
420  allocated.size = bytes;
421  allocated.pointer = m_upstream->do_allocate(bytes, m_options.alignment);
422  m_allocated.push_back(allocated);
423  bucket.previous_allocated_count = n;
424 
425  for (std::size_t i = 0; i < n; ++i)
426  {
427  bucket.free_blocks.push_back(
428  static_cast<void_ptr>(
429  static_cast<char_ptr>(allocated.pointer) + i * bucket_size
430  )
431  );
432  }
433  }
434 
435  // allocate a block from the front of the bucket's free list
436  void_ptr ret = bucket.free_blocks.back();
437  bucket.free_blocks.pop_back();
438  return ret;
439  }

441  virtual void do_deallocate(void_ptr p, std::size_t n, std::size_t alignment = THRUST_MR_DEFAULT_ALIGNMENT) THRUST_OVERRIDE
442  {
443  n = (std::max)(n, m_options.smallest_block_size);
444  assert(detail::is_power_of_2(alignment));
445 
446  // verify that the pointer is at least as aligned as claimed
447  assert(reinterpret_cast<detail::intmax_t>(detail::pointer_traits<void_ptr>::get(p)) % alignment == 0);
448 
449  // the deallocated block is oversized and/or overaligned
450  if (n > m_options.largest_block_size || alignment > m_options.alignment)
451  {
452  typename oversized_block_vector::iterator it = find_if(m_oversized.begin(), m_oversized.end(), equal_pointers(p));
453  assert(it != m_oversized.end());
454 
455  oversized_block_descriptor oversized = *it;
456 
457  if (m_options.cache_oversized)
458  {
459  typename oversized_block_vector::iterator position = lower_bound(m_cached_oversized.begin(), m_cached_oversized.end(), oversized);
460  m_cached_oversized.insert(position, oversized);
461  return;
462  }
463 
464  m_oversized.erase(it);
465 
466  m_upstream->do_deallocate(p, oversized.size, oversized.alignment);
467 
468  return;
469  }
470 
471  // push the block to the front of the appropriate bucket's free list
472  std::size_t n_log2 = thrust::detail::log2_ri(n);
473  std::size_t bucket_idx = n_log2 - m_smallest_block_log2;
474  pool & bucket = m_pools[bucket_idx];
475 
476  bucket.free_blocks.push_back(p);
477  }
