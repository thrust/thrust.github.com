42  void * do_allocate(std::size_t bytes, std::size_t alignment = THRUST_MR_DEFAULT_ALIGNMENT) override
43  {
44 #if defined(__cpp_aligned_new)
45  return ::operator new(bytes, std::align_val_t(alignment));
46 #else
47  // allocate memory for bytes, plus potential alignment correction,
48  // plus store of the correction offset
49  void * p = ::operator new(bytes + alignment + sizeof(std::size_t));
50  std::size_t ptr_int = reinterpret_cast<std::size_t>(p);
51  // calculate the offset, i.e. how many bytes of correction was necessary
52  // to get an aligned pointer
53  std::size_t offset = (ptr_int % alignment) ? (alignment - ptr_int % alignment) : 0;
54  // calculate the return pointer
55  char * ptr = static_cast<char *>(p) + offset;
56  // store the offset right after the actually returned value
57  std::size_t * offset_store = reinterpret_cast<std::size_t *>(ptr + bytes);
58  *offset_store = offset;
59  return static_cast<void *>(ptr);
60 #endif
61  }
62 
63  void do_deallocate(void * p, std::size_t bytes, std::size_t alignment = THRUST_MR_DEFAULT_ALIGNMENT) override
64  {
65 #if defined(__cpp_aligned_new)
66 # if defined(__cpp_sized_deallocation)
67  ::operator delete(p, bytes, std::align_val_t(alignment));
68 # else
69  (void)bytes;
70  ::operator delete(p, std::align_val_t(alignment));
71 # endif
72 #else
73  (void)alignment;
74  char * ptr = static_cast<char *>(p);
75  // calculate where the offset is stored
76  std::size_t * offset = reinterpret_cast<std::size_t *>(ptr + bytes);
77  // calculate the original pointer
78  p = static_cast<void *>(ptr - *offset);
79  ::operator delete(p);
80 #endif
81  }
